
CREATE TABLE WBUri(key BIGINT AUTO_INCREMENT PRIMARY KEY, uri VARCHAR(1024) NOT NULL, enabled INT NOT NULL, lastModified TIMESTAMP NOT NULL, httpOperation VARCHAR(20) NOT NULL, controllerClass VARCHAR(255), resourceType INT, resourceExternalKey VARCHAR(100), externalKey VARCHAR(100) NOT NULL);

CREATE INDEX WBuri_Index on WBUri (uri, enabled, lastModified, httpOperation, externalKey)

CREATE TABLE WBWEBPAGE(KEY BIGINT AUTO_INCREMENT PRIMARY KEY, NAME VARCHAR(1024) NOT NULL, LASTMODIFIED TIMESTAMP NOT NULL, HTMLSOURCE CLOB, ISTEMPLATESOURCE  INT,  CONTENTTYPE VARCHAR(255), EXTERNALKEY VARCHAR(100) NOT NULL, HASH BIGINT, PAGEMODELPROVIDER VARCHAR(1024));

CREATE INDEX WBWEBPAGE_INDEX on WBWEBPAGE (NAME, LASTMODIFIED, CONTENTTYPE, EXTERNALKEY)

CREATE TABLE WBWEBPAGEMODULE(KEY BIGINT AUTO_INCREMENT PRIMARY KEY, NAME VARCHAR(1024) NOT NULL, LASTMODIFIED TIMESTAMP NOT NULL, HTMLSOURCE CLOB, ISTEMPLATESOURCE  INT, EXTERNALKEY VARCHAR(100) NOT NULL);

CREATE INDEX WBWEBPAGEMODULE_INDEX on WBWEBPAGEMODULE (NAME, LASTMODIFIED, EXTERNALKEY)


CREATE TABLE WBPROJECT(KEY VARCHAR(1024) PRIMARY KEY, DEFAULTLANGUAGE VARCHAR(255) NOT NULL, LASTMODIFIED TIMESTAMP NOT NULL, SUPPORTEDLANGUAGES VARCHAR(1024) NOT NULL);



CREATE TABLE WBPARAMETER(KEY BIGINT AUTO_INCREMENT PRIMARY KEY, NAME VARCHAR(1024) NOT NULL, VALUE VARCHAR(2048), LASTMODIFIED TIMESTAMP NOT NULL, OVERWRITEFROMURL INT,  OWNEREXTERNALKEY VARCHAR(100), EXTERNALKEY VARCHAR(100) NOT NULL, LOCALETYPE INT);

CREATE INDEX WBPARAMETER_INDEX on WBPARAMETER (NAME, LASTMODIFIED, EXTERNALKEY)



CREATE TABLE WBMESSAGE(KEY BIGINT AUTO_INCREMENT PRIMARY KEY, NAME VARCHAR(1024) NOT NULL, VALUE CLOB, LASTMODIFIED TIMESTAMP NOT NULL, ISTRANSLATED INT,  LCID VARCHAR(100), EXTERNALKEY VARCHAR(100) NOT NULL);

CREATE INDEX WBMESSAGE_INDEX on WBMESSAGE (NAME, LASTMODIFIED, EXTERNALKEY)



CREATE TABLE WBFILE (KEY BIGINT AUTO_INCREMENT PRIMARY KEY, NAME VARCHAR(1024) NOT NULL, LASTMODIFIED TIMESTAMP NOT NULL, SIZE BIGINT,  HASH BIGINT, EXTERNALKEY VARCHAR(100) NOT NULL, FILENAME VARCHAR(1024), SHORTTYPE VARCHAR(100), CONTENTTYPE VARCHAR(100), ADJUSTEDCONTENTTYPE VARCHAR(100), BLOBKEY VARCHAR(2048), PUBLICURL VARCHAR(1024), THUMBNAILPUBLICURL VARCHAR(1024), THUMBNAILBLOBKEY VARCHAR(2048));

CREATE INDEX WBFILE_INDEX on WBFILE (NAME, LASTMODIFIED, EXTERNALKEY, CONTENTTYPE, SHORTTYPE, SIZE)


CREATE TABLE WBARTICLE (KEY BIGINT AUTO_INCREMENT PRIMARY KEY, TITLE VARCHAR(1024) NOT NULL, LASTMODIFIED TIMESTAMP NOT NULL, HTMLSOURCE CLOB, EXTERNALKEY VARCHAR(100) NOT NULL);

CREATE INDEX WBARTICLE_INDEX on WBARTICLE (TITLE, LASTMODIFIED, EXTERNALKEY)

CREATE TABLE WBResource(key VARCHAR(100) PRIMARY KEY, NAME VARCHAR(1024), type INT);
