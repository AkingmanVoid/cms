CREATE TABLE WPBUri(PRIVKEY BIGINT AUTO_INCREMENT PRIMARY KEY, uri VARCHAR(255) NOT NULL, enabled INT NOT NULL, lastModified TIMESTAMP NOT NULL, httpOperation VARCHAR(20) NOT NULL, controllerClass VARCHAR(255), resourceType INT, resourceExternalKey VARCHAR(100), externalKey VARCHAR(100) NOT NULL);

CREATE INDEX WPBuri_Index on WPBUri (uri, enabled, lastModified, httpOperation, externalKey);

CREATE TABLE WPBPAGE(PRIVKEY BIGINT AUTO_INCREMENT PRIMARY KEY, NAME VARCHAR(255) NOT NULL, LASTMODIFIED TIMESTAMP NOT NULL, HTMLSOURCE CLOB, ISTEMPLATESOURCE  INT,  CONTENTTYPE VARCHAR(255), EXTERNALKEY VARCHAR(100) NOT NULL, HASH BIGINT, PAGEMODELPROVIDER VARCHAR(1024));

CREATE INDEX WPBPAGE_INDEX on WPBPAGE (NAME, LASTMODIFIED, CONTENTTYPE, EXTERNALKEY);

CREATE TABLE WPBPAGEMODULE(PRIVKEY BIGINT AUTO_INCREMENT PRIMARY KEY, NAME VARCHAR(255) NOT NULL, LASTMODIFIED TIMESTAMP NOT NULL, HTMLSOURCE CLOB, ISTEMPLATESOURCE  INT, EXTERNALKEY VARCHAR(100) NOT NULL);

CREATE INDEX WPBPAGEMODULE_INDEX on WPBPAGEMODULE (NAME, LASTMODIFIED, EXTERNALKEY);

CREATE TABLE WPBPROJECT(PRIVKEY VARCHAR(255) PRIMARY KEY, DEFAULTLANGUAGE VARCHAR(255) NOT NULL, LASTMODIFIED TIMESTAMP NOT NULL, SUPPORTEDLANGUAGES VARCHAR(1024) NOT NULL);

CREATE TABLE WPBPARAMETER(PRIVKEY BIGINT AUTO_INCREMENT PRIMARY KEY, NAME VARCHAR(255) NOT NULL, VALUE VARCHAR(2048), LASTMODIFIED TIMESTAMP NOT NULL, OVERWRITEFROMURL INT,  OWNEREXTERNALKEY VARCHAR(100), EXTERNALKEY VARCHAR(100) NOT NULL, LOCALETYPE INT);

CREATE INDEX WPBPARAMETER_INDEX on WPBPARAMETER (NAME, LASTMODIFIED, EXTERNALKEY);

CREATE TABLE WPBMESSAGE(PRIVKEY BIGINT AUTO_INCREMENT PRIMARY KEY, NAME VARCHAR(255) NOT NULL, VALUE CLOB, LASTMODIFIED TIMESTAMP NOT NULL, ISTRANSLATED INT,  LCID VARCHAR(100), EXTERNALKEY VARCHAR(100) NOT NULL);

CREATE INDEX WPBMESSAGE_INDEX on WPBMESSAGE (NAME, LASTMODIFIED, EXTERNALKEY);

CREATE TABLE WPBFILE (PRIVKEY BIGINT AUTO_INCREMENT PRIMARY KEY, NAME VARCHAR(255) NOT NULL, LASTMODIFIED TIMESTAMP NOT NULL, SIZE BIGINT,  HASH BIGINT, EXTERNALKEY VARCHAR(100) NOT NULL, FILENAME VARCHAR(1024), SHORTTYPE VARCHAR(100), CONTENTTYPE VARCHAR(100), ADJUSTEDCONTENTTYPE VARCHAR(100), BLOBKEY VARCHAR(2048), PUBLICURL VARCHAR(1024), THUMBNAILPUBLICURL VARCHAR(1024), THUMBNAILBLOBKEY VARCHAR(2048));

CREATE INDEX WPBFILE_INDEX on WPBFILE (NAME, LASTMODIFIED, EXTERNALKEY, CONTENTTYPE, SHORTTYPE, SIZE);

CREATE TABLE WPBARTICLE (PRIVKEY BIGINT AUTO_INCREMENT PRIMARY KEY, TITLE VARCHAR(255) NOT NULL, LASTMODIFIED TIMESTAMP NOT NULL, HTMLSOURCE CLOB, EXTERNALKEY VARCHAR(100) NOT NULL);

CREATE INDEX WPBARTICLE_INDEX on WPBARTICLE (TITLE, LASTMODIFIED, EXTERNALKEY);

CREATE TABLE WPBResource (RKEY VARCHAR(255) PRIMARY KEY, NAME VARCHAR(255), type INT);
