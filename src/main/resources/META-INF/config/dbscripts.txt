CREATE TABLE WBUri(PRIVKEY BIGINT AUTO_INCREMENT PRIMARY KEY, uri VARCHAR(255) NOT NULL, enabled INT NOT NULL, lastModified TIMESTAMP NOT NULL, httpOperation VARCHAR(20) NOT NULL, controllerClass VARCHAR(255), resourceType INT, resourceExternalKey VARCHAR(100), externalKey VARCHAR(100) NOT NULL);

CREATE INDEX WBuri_Index on WBUri (uri, enabled, lastModified, httpOperation, externalKey);

CREATE TABLE WBWEBPAGE(PRIVKEY BIGINT AUTO_INCREMENT PRIMARY KEY, NAME VARCHAR(255) NOT NULL, LASTMODIFIED TIMESTAMP NOT NULL, HTMLSOURCE CLOB, ISTEMPLATESOURCE  INT,  CONTENTTYPE VARCHAR(255), EXTERNALKEY VARCHAR(100) NOT NULL, HASH BIGINT, PAGEMODELPROVIDER VARCHAR(1024));

CREATE INDEX WBWEBPAGE_INDEX on WBWEBPAGE (NAME, LASTMODIFIED, CONTENTTYPE, EXTERNALKEY);

CREATE TABLE WBWEBPAGEMODULE(PRIVKEY BIGINT AUTO_INCREMENT PRIMARY KEY, NAME VARCHAR(255) NOT NULL, LASTMODIFIED TIMESTAMP NOT NULL, HTMLSOURCE CLOB, ISTEMPLATESOURCE  INT, EXTERNALKEY VARCHAR(100) NOT NULL);

CREATE INDEX WBWEBPAGEMODULE_INDEX on WBWEBPAGEMODULE (NAME, LASTMODIFIED, EXTERNALKEY);

CREATE TABLE WBPROJECT(PRIVKEY VARCHAR(255) PRIMARY KEY, DEFAULTLANGUAGE VARCHAR(255) NOT NULL, LASTMODIFIED TIMESTAMP NOT NULL, SUPPORTEDLANGUAGES VARCHAR(1024) NOT NULL);

CREATE TABLE WBPARAMETER(PRIVKEY BIGINT AUTO_INCREMENT PRIMARY KEY, NAME VARCHAR(255) NOT NULL, VALUE VARCHAR(2048), LASTMODIFIED TIMESTAMP NOT NULL, OVERWRITEFROMURL INT,  OWNEREXTERNALKEY VARCHAR(100), EXTERNALKEY VARCHAR(100) NOT NULL, LOCALETYPE INT);

CREATE INDEX WBPARAMETER_INDEX on WBPARAMETER (NAME, LASTMODIFIED, EXTERNALKEY);

CREATE TABLE WBMESSAGE(PRIVKEY BIGINT AUTO_INCREMENT PRIMARY KEY, NAME VARCHAR(255) NOT NULL, VALUE CLOB, LASTMODIFIED TIMESTAMP NOT NULL, ISTRANSLATED INT,  LCID VARCHAR(100), EXTERNALKEY VARCHAR(100) NOT NULL);

CREATE INDEX WBMESSAGE_INDEX on WBMESSAGE (NAME, LASTMODIFIED, EXTERNALKEY);

CREATE TABLE WBFILE (PRIVKEY BIGINT AUTO_INCREMENT PRIMARY KEY, NAME VARCHAR(255) NOT NULL, LASTMODIFIED TIMESTAMP NOT NULL, SIZE BIGINT,  HASH BIGINT, EXTERNALKEY VARCHAR(100) NOT NULL, FILENAME VARCHAR(1024), SHORTTYPE VARCHAR(100), CONTENTTYPE VARCHAR(100), ADJUSTEDCONTENTTYPE VARCHAR(100), BLOBKEY VARCHAR(2048), PUBLICURL VARCHAR(1024), THUMBNAILPUBLICURL VARCHAR(1024), THUMBNAILBLOBKEY VARCHAR(2048));

CREATE INDEX WBFILE_INDEX on WBFILE (NAME, LASTMODIFIED, EXTERNALKEY, CONTENTTYPE, SHORTTYPE, SIZE);

CREATE TABLE WBARTICLE (PRIVKEY BIGINT AUTO_INCREMENT PRIMARY KEY, TITLE VARCHAR(255) NOT NULL, LASTMODIFIED TIMESTAMP NOT NULL, HTMLSOURCE CLOB, EXTERNALKEY VARCHAR(100) NOT NULL);

CREATE INDEX WBARTICLE_INDEX on WBARTICLE (TITLE, LASTMODIFIED, EXTERNALKEY);

CREATE TABLE WBResource (RKEY VARCHAR(255) PRIMARY KEY, NAME VARCHAR(255), type INT);
