CREATE TABLE WPBUri(uri VARCHAR(255) NOT NULL, enabled INT NOT NULL, lastModified TIMESTAMP NOT NULL, httpOperation VARCHAR(20) NOT NULL, controllerClass VARCHAR(255), resourceType INT, resourceExternalKey VARCHAR(100), externalKey VARCHAR(100) PRIMARY KEY);

CREATE INDEX WPBuri_Index on WPBUri (uri, enabled, lastModified, httpOperation);

CREATE TABLE WPBPAGE(NAME VARCHAR(255) NOT NULL, LASTMODIFIED TIMESTAMP NOT NULL, HTMLSOURCE CLOB, ISTEMPLATESOURCE  INT,  CONTENTTYPE VARCHAR(255), EXTERNALKEY VARCHAR(100) PRIMARY KEY, HASH BIGINT, PAGEMODELPROVIDER VARCHAR(1024));

CREATE INDEX WPBPAGE_INDEX on WPBPAGE (NAME, LASTMODIFIED, CONTENTTYPE);

CREATE TABLE WPBPAGEMODULE(NAME VARCHAR(255) NOT NULL, LASTMODIFIED TIMESTAMP NOT NULL, HTMLSOURCE CLOB, ISTEMPLATESOURCE  INT, EXTERNALKEY VARCHAR(100) PRIMARY KEY);

CREATE INDEX WPBPAGEMODULE_INDEX on WPBPAGEMODULE (NAME, LASTMODIFIED);

CREATE TABLE WPBPROJECT(EXTERNALKEY VARCHAR(100) PRIMARY KEY, DEFAULTLANGUAGE VARCHAR(255) NOT NULL, LASTMODIFIED TIMESTAMP NOT NULL, SUPPORTEDLANGUAGES VARCHAR(1024) NOT NULL);

CREATE TABLE WPBPARAMETER(NAME VARCHAR(255) NOT NULL, VALUE VARCHAR(2048), LASTMODIFIED TIMESTAMP NOT NULL, OVERWRITEFROMURL INT,  OWNEREXTERNALKEY VARCHAR(100), EXTERNALKEY VARCHAR(100) PRIMARY KEY, LOCALETYPE INT);

CREATE INDEX WPBPARAMETER_INDEX on WPBPARAMETER (NAME, LASTMODIFIED);

CREATE TABLE WPBMESSAGE(NAME VARCHAR(255) NOT NULL, VALUE CLOB, LASTMODIFIED TIMESTAMP NOT NULL, ISTRANSLATED INT,  LCID VARCHAR(100), EXTERNALKEY VARCHAR(100) NOT NULL);

CREATE INDEX WPBMESSAGE_INDEX on WPBMESSAGE (NAME, LASTMODIFIED);

CREATE TABLE WPBFILE (LASTMODIFIED TIMESTAMP NOT NULL, SIZE BIGINT,  HASH BIGINT, EXTERNALKEY VARCHAR(100) PRIMARY KEY, FILENAME VARCHAR(1024), CONTENTTYPE VARCHAR(100), ADJUSTEDCONTENTTYPE VARCHAR(100), BLOBKEY VARCHAR(2048), DIRECTORYFLAG INT NOT NULL, OWNEREXTKEY VARCHAR(100) NOT NULL);

CREATE INDEX WPBFILE_INDEX on WPBFILE (FILENAME, LASTMODIFIED, CONTENTTYPE, SIZE, OWNEREXTKEY);

CREATE TABLE WPBARTICLE (TITLE VARCHAR(255) NOT NULL, LASTMODIFIED TIMESTAMP NOT NULL, HTMLSOURCE CLOB, EXTERNALKEY VARCHAR(100) PRIMARY KEY);

CREATE INDEX WPBARTICLE_INDEX on WPBARTICLE (TITLE, LASTMODIFIED);

CREATE TABLE WPBResource (RKEY VARCHAR(255) PRIMARY KEY, NAME VARCHAR(255), type INT);

CREATE TABLE WPBUri(privkey BIGINT AUTO_INCREMENT PRIMARY KEY, uri VARCHAR(255) NOT NULL, enabled INT NOT NULL, lastModified TIMESTAMP NOT NULL, httpOperation VARCHAR(20) NOT NULL, controllerClass VARCHAR(255), resourceType INT, resourceExternalKey VARCHAR(100), externalKey VARCHAR(100) NOT NULL);

CREATE INDEX WPBuri_Index on WPBUri (uri, enabled, lastModified, httpOperation, externalKey);

CREATE TABLE WPBPage(PRIVKEY BIGINT AUTO_INCREMENT PRIMARY KEY, NAME VARCHAR(1024) NOT NULL, LASTMODIFIED TIMESTAMP NOT NULL, HTMLSOURCE LONGTEXT, ISTEMPLATESOURCE  INT,  CONTENTTYPE VARCHAR(255), EXTERNALKEY VARCHAR(100) NOT NULL, HASH BIGINT, PAGEMODELPROVIDER VARCHAR(1024));

CREATE INDEX WPBPAGE_INDEX on WPBPage (NAME, LASTMODIFIED, CONTENTTYPE, EXTERNALKEY);

CREATE TABLE WPBPageModule(PRIVKEY BIGINT AUTO_INCREMENT PRIMARY KEY, NAME VARCHAR(1024) NOT NULL, LASTMODIFIED TIMESTAMP NOT NULL, HTMLSOURCE LONGTEXT, ISTEMPLATESOURCE  INT, EXTERNALKEY VARCHAR(100) NOT NULL);

CREATE INDEX WPBPAGEMODULE_INDEX on WPBPageModule (NAME, LASTMODIFIED, EXTERNALKEY);


CREATE TABLE WPBProject(PRIVKEY VARCHAR(255) PRIMARY KEY, DEFAULTLANGUAGE VARCHAR(255) NOT NULL, LASTMODIFIED TIMESTAMP NOT NULL, SUPPORTEDLANGUAGES VARCHAR(1024) NOT NULL);



CREATE TABLE WPBParameter(PRIVKEY BIGINT AUTO_INCREMENT PRIMARY KEY, NAME VARCHAR(1024) NOT NULL, VALUE VARCHAR(2048), LASTMODIFIED TIMESTAMP NOT NULL, OVERWRITEFROMURL INT,  OWNEREXTERNALKEY VARCHAR(100), EXTERNALKEY VARCHAR(100) NOT NULL, LOCALETYPE INT);

CREATE INDEX WPBPARAMETER_INDEX on WPBParameter (NAME, LASTMODIFIED, EXTERNALKEY);



CREATE TABLE WPBMessage(PRIVKEY BIGINT AUTO_INCREMENT PRIMARY KEY, NAME VARCHAR(1024) NOT NULL, VALUE LONGTEXT, LASTMODIFIED TIMESTAMP NOT NULL, ISTRANSLATED INT,  LCID VARCHAR(100), EXTERNALKEY VARCHAR(100) NOT NULL);

CREATE INDEX WPBMESSAGE_INDEX on WPBMessage (NAME, LASTMODIFIED, EXTERNALKEY);



CREATE TABLE WPBFile (PRIVKEY BIGINT AUTO_INCREMENT PRIMARY KEY, LASTMODIFIED TIMESTAMP NOT NULL, SIZE BIGINT,  HASH BIGINT, EXTERNALKEY VARCHAR(100) NOT NULL, FILENAME VARCHAR(1024), CONTENTTYPE VARCHAR(100), ADJUSTEDCONTENTTYPE VARCHAR(100), BLOBKEY VARCHAR(2048), DIRECTORYFLAG INT NOT NULL, OWNEREXTKEY VARCHAR(100) NOT NULL);

CREATE INDEX WPBFILE_INDEX on WPBFile (FILENAME, LASTMODIFIED, EXTERNALKEY, CONTENTTYPE, SIZE, OWNEREXTKEY);


CREATE TABLE WPBArticle (PRIVKEY BIGINT AUTO_INCREMENT PRIMARY KEY, TITLE VARCHAR(1024) NOT NULL, LASTMODIFIED TIMESTAMP NOT NULL, HTMLSOURCE LONGTEXT, EXTERNALKEY VARCHAR(100) NOT NULL);

CREATE INDEX WPBARTICLE_INDEX on WPBArticle (TITLE, LASTMODIFIED, EXTERNALKEY);

CREATE TABLE WPBResource(RKEY VARCHAR(255) PRIMARY KEY, NAME VARCHAR(1024), type INT);
